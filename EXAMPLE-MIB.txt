-- Definition of a generic ExampleNotificationStatus type
ExampleNotificationStatus ::= TEXTUAL-CONVENTION
    STATUS current
    DESCRIPTION
        "Indicates the enabling or disabling of a particular class of
        notifications."
    SYNTAX INTEGER {
        disabled (0),   -- This class of notifications is disabled
        enabled  (1)    -- This class of notifications is enabled
}

-- Table
------------------------------------------------------------------------

exampleTableNumber OBJECT-TYPE
    SYNTAX      Unsigned32
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
        "The number of entries in the exampleTable."
    ::= { exampleTables 1 }

exampleTable OBJECT-TYPE
    SYNTAX      SEQUENCE OF ExampleTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
        "An example table"
    ::= { exampleTables 2 }

ExampleTableEntry ::=
    SEQUENCE {
        exampleTableEntryIndex  Unsigned32,
        exampleTableEntryDesc   DisplayString,
        exampleTableEntryValue  Integer32
    }

exampleTableEntry OBJECT-TYPE
    SYNTAX      ExampleTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
        "A particular exampleTable row."
    INDEX { exampleTableEntryIndex }
    ::= { exampleTable 1 }

exampleTableEntryIndex OBJECT-TYPE
    SYNTAX      Unsigned32
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
        "The index column used to generate numerical indices into
        exampleTable."
    ::= { exampleTableEntry 1 }

exampleTableEntryDesc OBJECT-TYPE
    SYNTAX      DisplayString
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
        "An exampleTableEntry's description."
    ::= { exampleTableEntry 2 }

exampleTableEntryValue OBJECT-TYPE
    SYNTAX      Integer32
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
        "An exampleTableEntry's value."
    ::= { exampleTableEntry 4 }

-- Notifications

events      OBJECT IDENTIFIER ::= { exampleMIBNotifications 0 }
operation   OBJECT IDENTIFIER ::= { exampleMIBNotifications 1 }

exampleUnsignedROChange NOTIFICATION-TYPE
    OBJECTS {
        exampleUnsignedRO
    }
    STATUS  current
    DESCRIPTION
        "An exampleScalarsChange notification signifies that there has
        been a change to the value of exampleUnsignedRO."
    ::= { events 1 }

exampleTableChange NOTIFICATION-TYPE
    OBJECTS {
        exampleTableEntryDesc, exampleTableEntryValue
    }
    STATUS  current
    DESCRIPTION
        "An exampleTableChange notification signifies that there has
        been a change to an exampleTableEntry."
    ::= { events 2 }

    SYNTAX      ExampleNotificationStatus

    STATUS      current
    DESCRIPTION

        enabled or disabled."
